**Step 1: Add client-side fetch helpers to api.ts**

Add two new client-side fetch functions to `apps/web/src/lib/api.ts`. These use the browser Supabase client (not the server one used by existing `fetchWithAuth`). Add them at the end of the file:

```typescript
// --- Client-side fetch helpers (for use in 'use client' components) ---
import { createClient as createBrowserSupabaseClient } from '@/lib/supabase/client'

export async function fetchOrderById(id: number): Promise<Order> {
  const supabase = createBrowserSupabaseClient()
  const {
    data: { session },
  } = await supabase.auth.getSession()
  const response = await fetch(`${API_BASE_URL}/api/orders/${id}`, {
    headers: {
      'Content-Type': 'application/json',
      ...(session?.access_token ? { Authorization: `Bearer ${session.access_token}` } : {}),
    },
  })
  if (!response.ok) throw new Error(`API error: ${response.status}`)
  return response.json()
}

export async function fetchProductById(id: number): Promise<Product> {
  const supabase = createBrowserSupabaseClient()
  const {
    data: { session },
  } = await supabase.auth.getSession()
  const response = await fetch(`${API_BASE_URL}/api/products/${id}`, {
    headers: {
      'Content-Type': 'application/json',
      ...(session?.access_token ? { Authorization: `Bearer ${session.access_token}` } : {}),
    },
  })
  if (!response.ok) throw new Error(`API error: ${response.status}`)
  return response.json()
}
```

**CRITICAL:** The existing `import { createClient } from '@/lib/supabase/server'` at the top is for Server Components. The new functions use `createClient` from `@/lib/supabase/client` which is the browser client. Use a renamed import alias (`createBrowserSupabaseClient`) to avoid name collision with the existing server import.

**IMPORTANT:** The `API_BASE_URL` constant already exists at line 9 of api.ts. Reuse it — do not redeclare.
