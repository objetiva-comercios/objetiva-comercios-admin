export interface Inventario {
  id: number
  nombre: string
  fecha: string
  depositoId: number
  depositoNombre?: string // joined from depositos table
  descripcion: string | null
  estado: 'pendiente' | 'en_curso' | 'finalizado' | 'cancelado'
  totalArticulos?: number // count summary
  createdAt: string
  updatedAt: string
}

export interface InventarioArticulo {
id: number

articuloCodigo: string
cantidadContada: number

createdAt: string
updatedAt: string
}

export interface InventarioArticuloWithDiscrepancy extends InventarioArticulo {

diferencia: number
}

export interface InventarioSector {

depositoId: number
nombre: string
columnas: string[]
createdAt: string
updatedAt: string
}

**api.ts** — add server-side fetch functions (add imports for new types at top):

- `fetchInventarios(params?: { page?, limit?, estado?, fechaDesde?, fechaHasta? }): Promise<PaginatedResponse<Inventario>>` — builds query string, calls fetchWithAuth
- `fetchInventario(id: number): Promise<Inventario>` — single event detail
- `fetchInventarioArticulos(inventarioId: number): Promise<InventarioArticuloWithDiscrepancy[]>` — counted articulos with discrepancy
- `fetchDispositivos(): Promise<DispositivoMovil[]>` — list all devices
- `fetchSectores(depositoId: number): Promise<InventarioSector[]>` — sectors for a deposito

**api.client.ts** — add client-side mutation functions (add imports for new types):

- `createInventario(data: { nombre, fecha, depositoId, descripcion? }): Promise<Inventario>`
- `updateInventario(id: number, data: { nombre?, descripcion? }): Promise<Inventario>`
- `transitionInventarioEstado(id: number, estado: string): Promise<Inventario>`
- `addInventarioArticulo(inventarioId: number, data: { articuloCodigo, cantidadContada?, dispositivoId?, sectorId?, observaciones? }): Promise<InventarioArticulo>`
- `updateInventarioArticulo(inventarioId: number, articuloId: number, data: { cantidadContada?, observaciones? }): Promise<InventarioArticulo>`
- `removeInventarioArticulo(inventarioId: number, articuloId: number): Promise<void>`
- `createDispositivo(data: { nombre, identificador, descripcion? }): Promise<DispositivoMovil>`
- `updateDispositivo(id: number, data: { nombre?, identificador?, descripcion? }): Promise<DispositivoMovil>`
- `toggleDispositivo(id: number): Promise<DispositivoMovil>`
- `createSector(depositoId: number, data: { nombre, columnas? }): Promise<InventarioSector>`
- `updateSector(depositoId: number, sectorId: number, data: { nombre?, columnas? }): Promise<InventarioSector>`
- `deleteSector(depositoId: number, sectorId: number): Promise<void>`
- `fetchSectoresClient(depositoId: number): Promise<InventarioSector[]>` — client-side fetch for sectores (used by depositos-list)

All mutations follow existing pattern: getAuthHeaders -> fetch with Content-Type+auth -> throwIfError -> return response.json().
For DELETE methods, don't return .json() (void return).
For `transitionInventarioEstado`, endpoint is `PATCH /api/inventarios/${id}/estado` with body `{ estado }`.
