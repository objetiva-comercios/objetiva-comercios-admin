```typescript
import { fetchInventarios, fetchDepositos } from '@/lib/api'
import { InventarioList } from '@/components/inventarios/inventario-list'

export default async function InventariosPage({
  searchParams,
}: {
  searchParams: { page?: string; limit?: string; estado?: string }
}) {
  const [inventarios, depositos] = await Promise.all([
    fetchInventarios({
      page: Number(searchParams.page) || 1,
      limit: Number(searchParams.limit) || 10,
      estado: searchParams.estado,
    }),
    fetchDepositos(),
  ])

  return <InventarioList inventarios={inventarios} depositos={depositos} />
}
```

**page.tsx** — server component:

```typescript
import { fetchInventario, fetchSectores } from '@/lib/api'
import { InventarioDetail } from '@/components/inventarios/inventario-detail'
import { notFound } from 'next/navigation'

export default async function InventarioDetailPage({
  params,
}: {
  params: { id: string }
}) {
  const id = Number(params.id)
  if (isNaN(id)) notFound()

  const inventario = await fetchInventario(id)
  const sectores = await fetchSectores(inventario.depositoId)

  return <InventarioDetail inventario={inventario} sectores={sectores} />
}
```
